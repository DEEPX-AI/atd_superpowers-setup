function createDb(path) {

      id         INTEGER PRIMARY KEY AUTOINCREMENT,
      email      TEXT    UNIQUE NOT NULL,

const jwt = require('jsonwebtoken');

function signToken(payload) {
  return jwt.sign(payload, process.env.JWT_SECRET, { expiresIn: '1h' });
}

function verifyToken(token) {
  return jwt.verify(token, process.env.JWT_SECRET);
}

module.exports = { signToken, verifyToken };

function createUserRepository(db) {
  const stmtFindByEmail = db.prepare('SELECT * FROM users WHERE email = ?');
  const stmtFindById = db.prepare('SELECT id, email, created_at FROM users WHERE id = ?');
  const stmtInsert = db.prepare('INSERT INTO users (email, password) VALUES (?, ?)');

  return {
    findByEmail(email) {
      return stmtFindByEmail.get(email) ?? null;
    },
    findById(id) {
      return stmtFindById.get(id) ?? null;
    },
    create(email, password) {
      const result = stmtInsert.run(email, password);
      return stmtFindById.get(result.lastInsertRowid);
    },
  };
}

module.exports = { createUserRepository };
```

function createHttpError(status, message) {
  const err = new Error(message);
  err.status = status;
  return err;
}

function createAuthService(userRepo, jwt, hash) {
  return {
    async register(email, password) {
      const existing = userRepo.findByEmail(email);
      if (existing) throw createHttpError(409, '이미 사용 중인 이메일입니다.');
      const hashed = await hash.hashPassword(password);
      const user = userRepo.create(email, hashed);
      return { id: user.id, email: user.email };
    },

    async login(email, password) {
      const user = userRepo.findByEmail(email);
      if (!user) throw createHttpError(401, '이메일 또는 비밀번호가 올바르지 않습니다.');
      const valid = await hash.comparePassword(password, user.password);
      if (!valid) throw createHttpError(401, '이메일 또는 비밀번호가 올바르지 않습니다.');
      const token = jwt.signToken({ sub: user.id, email: user.email });
      return { token };
    },
  };
}

module.exports = { createAuthService };
```

function createAuthenticateMiddleware(jwt) {
  return function authenticate(req, res, next) {
    const authHeader = req.headers.authorization;
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      const err = new Error('인증 토큰이 필요합니다.');
      err.status = 401;
      return next(err);
    }
    const token = authHeader.slice(7);
    try {
      req.user = jwt.verifyToken(token);
      next();
    } catch {
      const err = new Error('유효하지 않은 토큰입니다.');
      err.status = 401;
      next(err);
    }
  };
}

module.exports = { createAuthenticateMiddleware };

function isValidEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

function createAuthController(authService, userRepo) {
  return {
    async register(req, res, next) {
      try {
        const { email, password } = req.body ?? {};
        if (!email || !password) {
          const err = new Error('email과 password는 필수입니다.');
          err.status = 400;
          return next(err);
        }
        if (!isValidEmail(email)) {
          const err = new Error('유효한 이메일 형식이 아닙니다.');
          err.status = 400;
          return next(err);
        }
        if (password.length < 8) {
          const err = new Error('비밀번호는 최소 8자 이상이어야 합니다.');
          err.status = 400;
          return next(err);
        }
        const user = await authService.register(email, password);
        res.status(201).json(user);
      } catch (err) {
        next(err);
      }
    },

    async login(req, res, next) {
      try {
        const { email, password } = req.body ?? {};
        if (!email || !password) {
          const err = new Error('email과 password는 필수입니다.');
          err.status = 400;
          return next(err);
        }
        const result = await authService.login(email, password);
        res.status(200).json(result);
      } catch (err) {
        next(err);
      }
    },

    me(req, res, next) {
      const user = userRepo.findById(req.user.sub);
      if (!user) {
        const err = new Error('사용자를 찾을 수 없습니다.');
        err.status = 401;
        return next(err);
      }
      res.json({ id: user.id, email: user.email, created_at: user.created_at });
    },
  };
}

module.exports = { createAuthController };

const express = require('express');
const { createDb } = require('./db');
const { createUserRepository } = require('./repositories/user.repository');
const { createAuthService } = require('./services/auth.service');
const { createAuthController } = require('./controllers/auth.controller');
const { createAuthRouter } = require('./routes/auth.routes');
const { createAuthenticateMiddleware } = require('./middlewares/authenticate');
const jwt = require('./utils/jwt');
const hash = require('./utils/hash');

function createApp({ db, jwtSecret } = {}) {
  if (jwtSecret) process.env.JWT_SECRET = jwtSecret;
  if (!process.env.JWT_SECRET) {
    throw new Error('JWT_SECRET 환경 변수가 설정되지 않았습니다.');
  }

  const appDb = db || createDb(process.env.DB_PATH || './data/app.db');
  const userRepo = createUserRepository(appDb);
  const authService = createAuthService(userRepo, jwt, hash);
  const authenticate = createAuthenticateMiddleware(jwt);
  const authController = createAuthController(authService, userRepo);
  const authRouter = createAuthRouter(authController, authenticate);

  const app = express();
  app.use(express.json());
  app.use('/api/auth', authRouter);

  // 전역 에러 핸들러 (시그니처 4개 인자 필수)
  app.use((err, req, res, next) => {
    const status = err.status || 500;
    const message = err.message || '서버 오류가 발생했습니다.';
    if (status === 500 && process.env.NODE_ENV !== 'production') {
      console.error(err);
    }
    res.status(status).json({ error: message });
  });

  return app;
}

module.exports = { createApp };
